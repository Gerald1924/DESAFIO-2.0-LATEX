\section{ECUACIONES}

\begin{align*}

\text{Si} &  \quad x+3=5, \\
\texto{Entonces} & \quad (x+3)+(-3)=5+(-3);  \\
\texto{De donde:} & \quad x+(3+(-3))=5-3=2;  \\
\texto{De donde:} & \quad x+0=2;  \\
\texto{De donde:} & \quad x=2. \\

\end{align*}
